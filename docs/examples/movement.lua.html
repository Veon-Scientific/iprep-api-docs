movement.lua

--[[
Moves to a position in zone 2 and zone 3 and touches each
using the gorel command.
]]

veon.go("Zone2", "A1")

-- touch down
veon.gorel(0, 0, 20)
veon.gorel(0, 0, -20)

veon.go("Zone3", "C6")

-- touch down
veon.gorel(0, 0, 20)
veon.gorel(0, 0, -20)

-- return to safe position
veon.goabs(0, 20, 0)
